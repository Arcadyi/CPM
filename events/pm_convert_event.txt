namespace = pm_convert_event


pm_convert_event.1 = {
	type = country_event
	hidden = yes

	trigger = { is_player = yes }

	immediate = {
		every_scope_state = {
			limit = {
				NOT = { has_variable = pm_convert_origin }
			}
			set_variable = pm_convert_origin
		}
	}
}

pm_convert_event.2 = {
	type = country_event
	hidden = yes

	trigger = { is_player = yes }

	immediate = {
		every_scope_state = {
			limit = {
				NOT = { has_variable = pm_convert_origin }
			}
			pm_convert_custom = yes		
		}
	}
}
