    flavor = hungarian_revolution.2.f

    event_image = {
        video = "gfx/event_pictures/europenorthamerica_political_extremism.bk2"
    }

    icon = "gfx/interface/icons/event_icons/event_portrait.dds"
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

    duration = 5
    trigger = {
        has_variable = hungarian_revolution_begun
        NOT = {
            has_variable = asked_russia_for_help
        }
    }
   immediate = {
    c:RUS = {
        save_scope_as = russia_scope
    }
   }
   option = {
        name = hungarian_revolution.2.a
        custom_tooltip = askingrussiahungary
        hidden_effect = {
            c:RUS = {
                trigger_event = hungarian_revolution.3
            }
        } 
   }
   option = {
    name = hungarian_revolution.2.b
    }
}

            }

                country = c:RUS

            }
        }
            hidden_effect = {
                any_diplomatic_play = {
                    c:AUS = {
                        is_diplomatic_play_enemy_of = c:HUN      
                    }
                    add_initiator_backers = c:RUS
                }
            }
            ai_chance = {
                base = 10
                 modifier = {
                     trigger = {
                             c:AUS = {
                                 relations:root > 30 
                             }
                     }
                     add = 50
                }
          }
        }

        option = {
            name = hungarian_revolution.3.b
            custom_tooltip = russiarejectsshelp
            c:AUS = {
                change_relations = {
                    country = c:RUS
