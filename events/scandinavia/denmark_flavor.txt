denmark_flavor.4 = {
	type = country_event
	
	title = denmark_flavor.4.t
	desc = denmark_flavor.4.d
	flavor = denmark_flavor.4.f
	
	event_image = {
		video = "gfx/event_pictures/unspecific_politicians_arguing.bk2"
	}
	
	icon = "gfx/interface/icons/event_icons/event_portrait.dds"
	
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/asia/politician_parliament_motion"
	
	duration = 3
	
	trigger = {
		any_scope_character = {
			is_ruler = yes
			has_trait = imperious	# Just to make sure it's him, unlikely to occur otherwise
		}
	}
	cooldown = { months = normal_modifier_time }
	
	immediate = {
		random_scope_character = {
			limit = {
				is_ruler = yes
			}
			save_scope_as = frederik_the_monarch
		}
	}
	option = {
		name = denmark_flavor.4.a
		default_option = yes
		scope:frederik_the_monarch = {
			kill_character = yes
		}
		if = {
			limit = { c:HOL = { is_subject_of = c:DEN } }
			c:HOL = {
				change_relations = {
					country = c:DEN
					value = -10
				}
				change_relations = {
					country = c:PRU
					value = 10
				}
			}
		}
		if = {
			limit = { c:SCH = { is_subject_of = c:DEN } }
			c:SCH = {
				change_relations = {
					country = c:DEN
					value = -10
				}
				change_relations = {
					country = c:PRU
					value = 10
				}
			}
		}
		c:DEN = {
			create_character = {
				first_name = Frederick
				last_name = Christian
				heir = yes
				noble = yes
				birth_date = 1808.10.6
				interest_group = ig_landowners
				ideology = ideology_moderate
				traits = {
					tactful
					charismatic
					hedonist
				}
				#on_created = {
				#	set_variable = is_lovenskiold
				#}
			}
		}
	}
}

# First Schleswig War
denmark_flavor.5 = {
	type = country_event
	
	title = denmark_flavor.5.t
	desc = denmark_flavor.5.d
	flavor = denmark_flavor.5.f
	
	event_image = {
		video = "gfx/event_pictures/unspecific_politicians_arguing.bk2"
	}
	
	icon = "gfx/interface/icons/event_icons/event_portrait.dds"
	
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/asia/politician_parliament_motion"
	
	duration = 3
	
	cooldown = { months = normal_modifier_time }
	
	# If there was just a freaking way to have a country properly join a diplomatic play
	immediate = {
		make_independent = yes
		c:SCH = {
			make_independent = yes
			if = {
				limit = { is_ai = yes }
				c:HOL = {
					create_diplomatic_pact = {
						country = c:SCH
						type = personal_union
					}
				}
			}
			else = {	# Less functional but nonetheless working
				create_diplomatic_play = {
					name = first_schleswig_war
					target_country = c:DEN
					type = dp_independence
				}
			}
		}
		create_diplomatic_play = {
			name = first_schleswig_war
			target_country = c:DEN
			type = dp_independence
		}
		random_diplomatic_play = {
			limit = {
				any_scope_play_involved = {
					THIS = ROOT
				}
			}
			save_scope_as = cur_play
		}
		scope:cur_play = {
			# There is no 'restore union' one sadly
			# Fix with event at later date
			add_war_goal = {
				holder = c:DEN
				target_country = c:SCH
				type = make_dominion
			}
			add_war_goal = {
				holder = c:DEN
				target_country = c:HOL
				type = make_dominion
			}
		}
		if = {
			limit = { c:SCH = { is_ai = yes } }
			scope:cur_play = {
				add_war_goal = {
					holder = c:HOL
					type = conquer_state
					target_state = s:STATE_SCHLESWIG_HOLSTEIN.region_state:SCH
				}
			}
		}
		c:PRU = {	# Should help
			set_secret_goal = {
				country = c:HOL
				secret_goal = protect
			}
			set_secret_goal = {
				country = c:SCH
				secret_goal = protect
			}
			set_secret_goal = {
				country = c:DEN
				secret_goal = antagonize
			}
		}
	}
	
	option = {
		name = denmark_flavor.5.a
		default_option = yes
		if = {
			limit = {
				year < 1865
			}
			c:HOL = {
				create_character = {
					first_name = Frederick
					last_name = August
					culture = cu:danish
					ruler = yes
					noble = yes
					birth_date = 1808.8.23
					interest_group = ig_landowners
					ideology = ideology_royalist
					traits = {
						reckless
						basic_offensive_planner
					}
					on_created = {
						set_variable = is_married
					}
				}
			}
		}
		else_if = {	# Hopefully won't happen later than this but fallback
			limit = {
				year < 1869
			}
			c:HOL = {
				create_character = {
					first_name = Christian
					last_name = August
					culture = cu:danish
					ruler = yes
					noble = yes
					birth_date = 1829.7.6
					interest_group = ig_landowners
					ideology = ideology_moderate
					traits = {
						romantic
					}
					on_created = {
						set_variable = is_married
					}
				}
			}
		}
		show_as_tooltip = {
			create_diplomatic_play = {
				name = first_schleswig_war
				target_country = c:DEN
				type = dp_independence
			}
		}
		# Add alternative later
		add_loyalists = {
			culture = cu:north_german
			value = 0.25
		}
		change_relations = {
			country = c:DEN
			value = -50
		}
		c:SCH = {
			if = {
				limit = { is_ai = no }
				trigger_event = { id = denmark_flavor.6 }
			}
			else = {
				change_relations = {
					country = c:DEN
					value = -50
				}
			}
		}
	}
}
