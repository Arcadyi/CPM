# Problems Between Swedes & Finnish
finland_flavor.3 = {
	type = country_event
	
	title = finland_flavor.3.t
	desc = finland_flavor.3.d
	flavor = finland_flavor.3.f
	
	event_image = {
		video = "gfx/event_pictures/europenorthamerica_rich_and_poor.bk2"
	}
	
	icon = "gfx/interface/icons/event_icons/event_portrait.dds"
	
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/asia/politician_parliament_motion"
	
	duration = 3
	
	cooldown = { months = short_modifier_time }
	
	trigger = {
		has_modifier = finland_our_language_outside_courts
		any_scope_state = {
			is_incorporated = yes
			any_scope_pop = {
				culture = cu:swedish
				strata = middle
			}
			any_scope_pop = {
				culture = cu:finnish
			}
		}
	}
	
	immediate = {
		random_scope_state = {
			limit = {
				is_incorporated = yes
				any_scope_pop = {
					culture = cu:swedish
					strata = middle
				}
				any_scope_pop = {
					culture = cu:finnish
				}
			}
			save_scope_as = the_conflict_state
		}
	}
	
	option = {
		name = finland_flavor.3.a
		default_option = yes
		scope:the_conflict_state = {
			add_radicals = {
				culture = cu:swedish
				strata = middle
				value = 0.1
			}
		}
	}
	option = {
		name = finland_flavor.3.b
		ai_chance = {
			base = 0
		}
		change_variable = {
			name = finland_language_debate_loss_var
			add = 1
		}
		scope:the_conflict_state = {
			add_radicals = {
				culture = cu:finnish
				value = 0.05
			}
		}
	}
}

finland_flavor.4 = {
	type = country_event
	
	title = finland_flavor.4.t
	desc = finland_flavor.4.d
	flavor = finland_flavor.4.f
	
	event_image = {
		video = "gfx/event_pictures/unspecific_politicians_arguing.bk2"
	}
	
	icon = "gfx/interface/icons/event_icons/event_portrait.dds"
	
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/asia/politician_parliament_motion"
	
	duration = 3
	
	cooldown = { months = normal_modifier_time }
	
	trigger = {
		any_scope_state = {
			is_incorporated = yes
			any_scope_pop = {
				is_pop_type = academics
			}
		}
		NOT = { has_modifier = finland_swedish_oppose_the_academics }
	}
	
	option = {
		name = finland_flavor.4.a
		default_option = yes
		ai_chance = {
			base = 9
		}
		add_modifier = {
			name = finland_swedish_oppose_the_academics
			months = short_modifier_time
		}
		add_radicals = {
			pop_type = academics
			value = 0.05
		}
	}
	option = {
		name = finland_flavor.4.b
		ai_chance = {
			base = 0
			modifier = {
				trigger = {
					authority >= 25
				}
				add = 1
			}
		}
		change_variable = {
			name = finland_language_debate_loss_var
			add = 1
		}
	}
}
