accept_bosniaks = {
	is_shown = {
		exists = c:YUG
		c:YUG = THIS
any_scoped_state = {
   limit = {
   has_homeland = cu:bosniak
           }
       }
	}
	
	possible = {	
		NOT = { 
		        has_law = law_type:law_state_religion
				culture_accepted = cu:bosniak
				}
        has_law = law_type:law_cultural_exclusion
		has_law = law_type:law_total_separation
	}
	
	when_taken = {
        add_primary_culture = cu:bosniak
	}

	ai_chance = {
		base = 65
	}
}
