            has_trait = academic
            has_trait = banker
            has_trait = bureaucrat
            has_trait = clerk

            has_trait = police_chief
            has_trait = political_dynasty
            has_trait = religious_leader
            has_trait = soldier
            has_trait = teacher

        age > define:NPortrait|GRACEFUL_AGING_START
