staff_officer_origin_commander = {
	type = skill
	texture = "gfx/interface/icons/character_trait_icons/unused/engineer.dds"
	
	command_modifier = {
		unit_offense_add = 10
		unit_defense_add = 10
		unit_supply_consumption_mult = -0.1
	}
	
	possible = {
		has_role = general
		owner = {
			has_technology_researched = general_staff
		}
		NOT = {
			has_trait = bandit
		}
	}
	
	weight = {
		value = 5
		if = {
			limit = { 
				owner = {
					any_scope_building = {
						is_building_type = building_stuff_headquarters
						has_active_production_method = pm_professional_staff_system
					}
				}
			}
			multiply = 5
		}
		else_if = {
			limit = { 
				owner = {
					any_scope_building = {
						is_building_type = building_stuff_headquarters
						has_active_production_method = pm_complete_analysis_staff_system
					}
				}
			}
			multiply = 7.5
		}
		else = {
			limit = { 
				owner = {
					any_scope_building = {
						is_building_type = building_stuff_headquarters
						has_active_production_method = pm_modern_command_system_staff_system
					}
				}
			}
			multiply = 10
		}
	}
	
	replace = {		
	}
	
	value = 1
}
