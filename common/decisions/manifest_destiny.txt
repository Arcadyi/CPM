			OR = {
			NOT = { has_variable = usa_western_purchase_var }
			}
		}
	}

	possible = {
		exists = c:USA
		root = c:USA
		c:USA = {

			NOT = { has_variable = mexican_american_war_treaty }
		}
		c:USA = {
			has_diplomatic_pact = {
   					who = c:MEX
   					type = dominion
   				}
		}
	}

	when_taken = {
		c:USA = { set_variable = mexican_american_war_treaty }

		set_variable = {
				name = gasden_purchase_var

			}	
		}

	ai_chance = {
		base = 100
	}
}


manifest_destiny_rebuked = { ##victory

	is_shown = {
		exists = c:MEX
		root = c:MEX
		c:USA = {

			NOT = { has_variable = mexican_american_war_treaty }
			NOT = { has_variable = mexican_american_war_treaty_loss }
		}
	}

	possible = {
		exists = c:MEX
		root = c:MEX
		is_diplomatic_play_committed_participant = no
 		is_at_war = no
		c:USA = {

			NOT = { has_variable = mexican_american_war_treaty }
			NOT = { has_variable = mexican_american_war_treaty_loss }
		}
	NOT = {	c:USA = {
			has_diplomatic_pact = {
   					who = c:MEX
   					type = dominion
   				}
		} }
	}

	when_taken = {
		if = { limit = { 
    		c:CAL = {
    		is_player = no
    		OR = {
    			is_subject = no
    			is_subject_of = c:MEX
    		}
			}
    		exists = c:CAL
    	}
    		annex = c:CAL
    	}
		c:USA = { set_variable = mexican_american_war_treaty_loss }
		c:MEX = { add_modifier = {
			name = manifest_destiny_rebuked_mod
			months = 250
			is_decaying = yes }}
		}

	ai_chance = {
		base = 100
	}
}
