gbr_expand_cape_colony = {
	is_shown = {
		this = c:GBR
		exists = c:SAF
		c:SAF = {
			is_subject_of = c:GBR
		}
		OR = { 
			any_state = {
				owner = c:GBR
				region = sr:region_southern_africa
			}
			any_subject_or_below = {
				capital.region = sr:region_southern_africa
				NOT = { 
					THIS = c:SAF
					is_subject_of = c:SAF
				}
			}
		}
	}
	possible = {
		OR = { 
			any_state = {
				owner = c:GBR
				region = sr:region_southern_africa
			}
			any_subject_or_below = {
				capital.region = sr:region_southern_africa
				NOT = { 
					THIS = c:SAF
					is_subject_of = c:SAF
				}
			}
		}
	}
	when_taken = {
		custom_tooltip = expand_cape_colony_tt
		if = {
			limit= {
				any_state = {
					owner = c:GBR
					region = sr:region_southern_africa
				}
			}
			every_scope_state = {
				limit = {
					owner = c:GBR
					region = sr:region_southern_africa
				}
				set_state_owner = c:SAF
				set_state_type = unincorporated
			}
		}
		if = {
			limit= {
				any_subject_or_below = {
					capital.region = sr:region_southern_africa
					NOT = { 
						THIS = c:SAF
						is_subject_of = c:SAF
					}
				}
			}
			every_subject_or_below = {
				if = {
					limit = { 
						is_subject_type = subject_type_protectorate
						capital.region = sr:region_southern_africa
						NOT = { 
							THIS = c:SAF
							is_subject_of = c:SAF
						} 						
					}
					hidden_effect = { make_independent = yes }
					save_scope_as = new_saf_protectorate
					c:SAF = {
						create_diplomatic_pact = {
							country = scope:new_saf_protectorate
							type = protectorate
						}
					}
				}
				if = {
					limit = { 
						is_subject_type = subject_type_vassal
						capital.region = sr:region_southern_africa
						NOT = { 
							THIS = c:SAF 
							is_subject_of = c:SAF 
						} 
					}
					hidden_effect = { make_independent = yes }
					save_scope_as = new_saf_vassal
					c:SAF = {
						create_diplomatic_pact = {
							country = scope:new_saf_vassal
							type = puppet
						}
					}
				}
				if = {
					limit = { 
						is_subject_type = subject_type_puppet
						capital.region = sr:region_southern_africa
						NOT = { 
							THIS = c:SAF
							is_subject_of = c:SAF
						} 
					}
					hidden_effect = { make_independent = yes }
					save_scope_as = new_saf_puppet
					c:SAF = {
						create_diplomatic_pact = {
							country = scope:new_saf_puppet
							type = puppet
						}
					}
				}
			}
		}
	}
	ai_chance = {
		base = 100
	}
}
