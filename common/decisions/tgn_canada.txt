tgn_canada_discuss_the_dominion = {
	is_shown = {
		has_modifier = canada_tgnxxx
		OR = {
			AND = {
				exists = c:NBS
				c:NBS = {
					is_player = no
				}
			}
			AND = {
				exists = c:NVS
				c:NVS = {
					is_player = no
				}
			}
		}
		NOT = { has_global_variable = tgn_charlottetown_conference_held }
		has_global_variable = tgn_can_now_proceed_with_the_dominion
	}

	possible = {
		trigger_if = {
			limit = {
				exists = c:PEI
			}
			c:PEI = {
				relations:root >= relations_threshold:amicable
				this.gdp < root.gdp
			}
		}
		trigger_if = {
			limit = {
				exists = c:NBS
			}
			c:NBS = {
				relations:root >= relations_threshold:amicable
				this.gdp < root.gdp
			}
		}
		trigger_if = {
			limit = {
				exists = c:NVS
			}
			c:NVS = {
				relations:root >= relations_threshold:amicable
				this.gdp < root.gdp
			}
		}
		custom_tooltip = {
			text = tgn_has_not_recently_expanded_tooltip
			NOT = { 
				has_modifier = canada_test_modifier
			}
		}
	}
	
	when_taken = {
		custom_tooltip = tgn_charlottetown_conference_will_be_held_tooltip
		hidden_effect = {
			trigger_event = { id = tgn_je_events.6 popup = yes }
			set_global_variable = {
				name = tgn_charlottetown_conference_held
				value = yes
			}
		}
	}
	
	ai_chance = {
		always = yes
	}
}
tgn_canada_rupert_land_act_decision = {
	is_shown = {
		has_modifier = canada_tgnxxx
		exists = c:HBC
		c:HBC = {
			is_player = no
		}
		NOT = { has_global_variable = tgn_canada_rupert_land_act_decision_taken }
		has_global_variable = tgn_stage1_of_the_dominion_completed
	}

	possible = {
		c:HBC = {
			relations:root >= relations_threshold:amicable
			this.gdp < root.gdp
		}
		c:GBR = {
			relations:root >= relations_threshold:amicable
		}
		custom_tooltip = {
			text = tgn_dominion_can_expand_tooltip
			has_global_variable = tgn_stage2_is_now_available
		}
		custom_tooltip = {
			text = tgn_has_not_recently_expanded_tooltip
			NOT = { 
				has_modifier = canada_test_modifier
			}
		}
	}
	when_taken = {
		custom_tooltip = tgn_will_send_the_crown_acquisition_request_HBC_tooltip
		hidden_effect = {
			add_journal_entry = {
				type = je_tgn_the_crown_considering_HBC_acquisition
			}
			set_global_variable = {
				name = tgn_canada_rupert_land_act_decision_taken
				value = yes
			}
			add_modifier = {
				name = canada_test_modifier
				months = 30
			}
		}
	}
	
	ai_chance = {
		always = yes
	}
}
