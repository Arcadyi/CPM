coa_no_NET_overlord_trigger = {
	exists = c:NET
	c:NET = { save_temporary_scope_as = custom_overlord }
	NOT = { has_game_rule = no_subject_flags }
	exists = scope:overlord
	NOT = {
		scope:overlord = {
			OR = {
				this = scope:custom_overlord
				any_overlord_or_above = {
					this = scope:custom_overlord
				} 
			}
		}
	}
}
