government_auto_expand_rule = { 
	is_building_type = building_government_administration
	state.tax_capacity < state.tax_capacity_usage
	building_has_goods_shortage = no
	owner.construction_queue_duration < max_autoexpand_queue_weeks
}

railway_auto_expand_rule = { 
	#cash_reserves_ratio > 0.75 #0.95
	is_building_type = building_railway
	state.relative_infrastructure < 1
	building_has_goods_shortage = no
	devastation < 1	
	owner.construction_queue_duration < max_autoexpand_queue_weeks
}

default_auto_expand_rule = { 
	cash_reserves_ratio > 0.95
	state.market_access > 0.5
	OR = {
		state.state_unemployment_rate > 0
	}
	owner.construction_queue_duration < max_autoexpand_queue_weeks
}
