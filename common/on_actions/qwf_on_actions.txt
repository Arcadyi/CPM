on_battle_started = {
	on_actions = {
		fort_effect_for_battle
		artillery_fire_support_for_battle
		advanced_ship_advantages_for_battle
	}
}

on_monthly_pulse_country = {
	on_actions = {
		remove_invalid_building_stuff_headquarters
	}
}

on_battle_ended = {
	on_actions = {
		#remove_advanced_ship_advantages_for_battle
	}
}

on_war_end = {
	on_actions = {
		#remove_artillery_fire_support_for_battle_country
	}
}

on_monthly_pulse_character = {
	on_actions = {
		#advanced_ship_advantages_for_battle
	}
}

fort_effect_for_battle = {
	effect = {
		if = {
			limit = {
				exists = scope:attacker
				scope:state = {
					NOT = { has_modifier = destroyed_fort }
					any_scope_building = {
						is_building_type = building_fortifications
					}
				}
				scope:state.qwf_state_fort_value < scope:attacker.qwf_army_attack_fort_value
			}
			scope:state = {
				add_modifier = {
					name = destroyed_fort
					months = 12
				}
			}
		}
	}
}

remove_invalid_building_stuff_headquarters = {
	effect = {
		if = {
			limit = {
				any_scope_state = {
					any_scope_building = {
						is_building_type = building_stuff_headquarters
						level >= 1
					}
					is_capital = no
				}
			}
			every_scope_state = {
				limit = {
					is_capital = no
				}
				remove_building = building_stuff_headquarters
			}
			post_notification = invalid_building_stuff_headquarters
		}
	}
}
