nor_the_great_transformation = {
	icon = "gfx/interface/icons/event_icons/event_industry.dds"

	is_shown_when_inactive = {
		exists = c:NOR
		c:NOR = THIS
	}
	
	possible = {
		has_industry_building = yes
	}
	
	complete = {
		any_scope_building = {
			OR = {
				is_building_type = building_food_industry
				is_building_type = building_textile_mills
				is_building_type = building_furniture_manufacturies
				is_building_type = building_glassworks
				is_building_type = building_paper_mills
			}
			occupancy > 0.9
			level >= 3
		}
		OR = {
			any_scope_building = {
				is_building_type = building_tooling_workshops
				occupancy > 0.9
				level >= 2
			}
			capital.market = {
				any_trade_route = {
					importer.owner = root
					goods = g:tools
				}
			}
		}
	}
	
	on_complete = {
		trigger_event = { id = norway_flavor.2 }
	}
	
	weight = 100
}

nor_the_merchant_navy = {
	icon = "gfx/interface/icons/event_icons/event_trade.dds"

	is_shown_when_inactive = {
		exists = c:NOR
		c:NOR = THIS
	}
	
	possible = {
		any_scope_building = {
			is_building_type = building_port
		}
	}
	
	complete = {
		in_default = no
		NOT = { has_law = law_type:law_isolationism }
		any_scope_building = {
			is_building_type = building_shipyards
			occupancy > 0.9
			level >= 2
		}
		any_scope_building = {	# Increase later?
			is_building_type = building_port
			NOT = { has_active_production_method = pm_anchorage }
			occupancy > 0.9
			level >= 5
		}
	}
	
	on_complete = {
		trigger_event = { id = norway_flavor.3 }
		show_as_tooltip = {
			add_modifier = {
				name = norway_the_great_merchant_navy
				months = -1
			}
		}
	}
	
	weight = 100
}
