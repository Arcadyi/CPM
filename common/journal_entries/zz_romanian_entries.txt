
# Draft the First Constitution
je_draft_constitution = {

	icon = "gfx/interface/icons/event_icons/tutorial_icon.dds"
	
	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
		has_technology_researched = egalitarianism
	}

	on_complete = {
		trigger_event = {
			id = romanian_events.2
			popup = no
		}

		add_journal_entry = {
			type = je_secure_dobruja
		}

		set_variable = drafted_constitution
	}
	
	current_value = {
		value = var:drafting_progress_var
	}
	
	goal_add_value = {
		add = 3
	}

	weight = 10000
	
	progressbar = yes

	should_be_pinned_by_default = yes
}

# Union of the Romanian Principalities
je_little_union = {

	icon = "gfx/interface/icons/event_icons/event_election.dds"
	
	complete = {
		OR = {
			hidden_trigger = {
				AND = {
					is_player = no
					game_date >= 1859
				}
			}
			AND = {
				is_at_war = no
				exists = c:MOL
				has_technology_researched = nationalism

        		NOR = {
					has_law = law_type:law_closed_borders
				}

				c:MOL = {
					relations:ROOT >= relations_threshold:amicable
				}

				c:RUS = {
					relations:ROOT >= relations_threshold:amicable
				}

				OR = {
					c:FRA = {
						relations:ROOT >= relations_threshold:amicable
					}
					c:GBR = {
						relations:ROOT >= relations_threshold:amicable
					}
				}

				OR = {
					c:AUS = {
						relations:ROOT >= relations_threshold:amicable
					}
					c:PRU = {
						relations:ROOT >= relations_threshold:amicable
					}
				}
			}
		}
	}

	on_complete = {
		trigger_event = {
			id = romanian_events.3
			popup = yes
		}

		change_variable = {
			name = drafting_progress_var
			add = 1
		}

		s:STATE_MOLDAVIA = {
			add_claim = ROOT
		}
	}

	weight = 9000

	should_be_pinned_by_default = yes
}
